% --- No \vfill before the very first section ---
\newif\iffirstsection
\firstsectiontrue

\newcommand{\cvsection}[1]{%
    \iffirstsection
        \firstsectionfalse
        \vspace{0.25em}
    \else
        \sectionspace
    \fi
    \noindent\textcolor{primary}{\large \textbf{#1}} \textcolor{primary}{\hrulefill}
    \par\vspace{0.3em}
}

\newcommand{\headerrow}[2]{
    \begin{tabular*}{\linewidth}{l@{\extracolsep{\fill}}r}
    #1 & \textcolor{primary}{#2} \\
    \end{tabular*}
}

% \cvheader{Job Title}
% Requires personal_data.tex to be \input before use.
\newcommand{\cvheader}[1]{%
    \begin{center}
        {\LARGE \textcolor{primary}{\textbf{\cvname}}}\\
        \vspace{0.05cm}
        \textcolor{primary}{\textbf{#1}}\\
        \vspace{0.1cm}
        \textcolor{primary}{\faMapMarker*} \ \cvlocation\ (\cvmobility) \quad|\quad
        \textcolor{primary}{\faEnvelope} \ \href{mailto:\cvemail}{\cvemail} \quad|\quad
        \textcolor{primary}{\faPhone} \ \cvphone \\
        \vspace{0.05cm}
        \textcolor{primary}{\faLinkedin} \ \href{https://\cvlinkedin}{\cvlinkedin} \quad|\quad
        \textcolor{primary}{\faGithub} \ \href{https://\cvgithub}{\cvgithub} \quad|\quad
        \textcolor{primary}{\faGlobe} \ \href{https://\cvwebsite}{\cvwebsite}
    \end{center}
}
